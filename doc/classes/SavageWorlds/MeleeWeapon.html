    def roll_damage(str, mod=0)
      str_roll = str.roll()
      dmg_roll = @damage.roll()

      t = str_roll.total + dmg_roll.total + mod
      result = MeleeResult.new(t, str_roll, dmg_roll)
      return result
    end
